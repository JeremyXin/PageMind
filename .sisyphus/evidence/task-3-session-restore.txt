Before:
```typescript
useEffect(() => {
  chatStorage.getActiveSession().then((session) => {
    if (session) {
      setMessages(session.messages);
      setSessionId(session.id);
    }
  });
  // ... rest of effect
}, [createPortListener]);
```

After:
```typescript
useEffect(() => {
  const initSession = async () => {
    const session = await chatStorage.getActiveSession();
    if (session) {
      setMessages(session.messages || []);
      setSessionId(session.id);
    }
  };
  initSession();
  // ... rest of effect
}, [createPortListener]);
```
